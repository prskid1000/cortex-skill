```python
from openpyxl import Workbook

wb = Workbook()
```

### Styled Report

```python
from openpyxl import Workbook
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side, numbers

wb = Workbook()
ws = wb.active
ws.title = "Monthly Report"

# -- Styles --
header_font = Font(name="Calibri", bold=True, color="FFFFFF", size=12)
header_fill = PatternFill(start_color="4472C4", end_color="4472C4", fill_type="solid")
header_alignment = Alignment(horizontal="center", vertical="center", wrap_text=True)
thin_border = Border(
    left=Side(style="thin"),
    right=Side(style="thin"),
    top=Side(style="thin"),
    bottom=Side(style="thin"),
)
currency_format = '"$"#,##0.00'
percent_format = "0.0%"

# -- Headers --
headers = ["Month", "Revenue", "Expenses", "Profit", "Margin"]
ws.append(headers)

for col_num, header in enumerate(headers, 1):
    cell = ws.cell(row=1, column=col_num)
    cell.font = header_font
    cell.fill = header_fill
    cell.alignment = header_alignment
    cell.border = thin_border

# -- Data --
data = [
    ["January", 52000, 38000],
    ["February", 48000, 35000],
    ["March", 61000, 42000],
    ["April", 55000, 39000],
    ["May", 67000, 44000],
    ["June", 72000, 48000],
]

for row_idx, (month, revenue, expenses) in enumerate(data, 2):
    ws.cell(row=row_idx, column=1, value=month).border = thin_border
    ws.cell(row=row_idx, column=2, value=revenue).border = thin_border
    ws.cell(row=row_idx, column=2).number_format = currency_format
    ws.cell(row=row_idx, column=3, value=expenses).border = thin_border
    ws.cell(row=row_idx, column=3).number_format = currency_format
    ws.cell(row=row_idx, column=4).value = revenue - expenses
    ws.cell(row=row_idx, column=4).number_format = currency_format
    ws.cell(row=row_idx, column=4).border = thin_border
    ws.cell(row=row_idx, column=5).value = (revenue - expenses) / revenue
    ws.cell(row=row_idx, column=5).number_format = percent_format
    ws.cell(row=row_idx, column=5).border = thin_border
    ws.cell(row=row_idx, column=5).alignment = Alignment(horizontal="center")

# -- Column widths --
ws.column_dimensions["A"].width = 14
ws.column_dimensions["B"].width = 16
ws.column_dimensions["C"].width = 16
ws.column_dimensions["D"].width = 16
ws.column_dimensions["E"].width = 12

# -- Totals row --
total_row = len(data) + 2
ws.cell(row=total_row, column=1, value="TOTAL").font = Font(bold=True)
for col in range(2, 5):
    col_letter = chr(64 + col)
    ws.cell(row=total_row, column=col).value = (
        f"=SUM({col_letter}2:{col_letter}{total_row - 1})"
    )
    ws.cell(row=total_row, column=col).number_format = currency_format
    ws.cell(row=total_row, column=col).font = Font(bold=True)
    ws.cell(row=total_row, column=col).border = Border(top=Side(style="double"))

wb.save("/tmp/styled_report.xlsx")
```

### Conditional Formatting

```python
from openpyxl import Workbook
from openpyxl.formatting.rule import (
    ColorScaleRule,
    DataBarRule,
    IconSetRule,
    CellIsRule,
    FormulaRule,
)
from openpyxl.styles import PatternFill, Font

wb = Workbook()
ws = wb.active
ws.title = "Conditional Formatting"

# Sample data
ws.append(["Name", "Score", "Status", "Trend"])
import random
random.seed(42)
for i in range(1, 21):
    ws.append([f"Item {i}", random.randint(10, 100), random.randint(1, 5), random.randint(-50, 50)])

# --- Color Scale (green-yellow-red) on Score column ---
ws.conditional_formatting.add(
    "B2:B21",
    ColorScaleRule(
        start_type="min", start_color="F8696B",   # red
        mid_type="percentile", mid_value=50, mid_color="FFEB84",  # yellow
        end_type="max", end_color="63BE7B",        # green
    ),
)

# --- Data Bars on Trend column ---
ws.conditional_formatting.add(
    "D2:D21",
    DataBarRule(
        start_type="min",
        end_type="max",
        color="638EC6",
        showValue=True,
    ),
)

# --- Icon Sets (3 arrows) on Status column ---
ws.conditional_formatting.add(
    "C2:C21",
    IconSetRule(
        icon_style="3Arrows",
        type="num",
        values=[1, 3, 4],
        showValue=True,
        reverse=False,
    ),
)

# --- Cell-Is Rule: highlight scores >= 80 ---
ws.conditional_formatting.add(
    "B2:B21",
    CellIsRule(
        operator="greaterThanOrEqual",
        formula=["80"],
        fill=PatternFill(start_color="C6EFCE", end_color="C6EFCE", fill_type="solid"),
        font=Font(color="006100", bold=True),
    ),
)

# --- Cell-Is Rule: highlight scores < 30 ---
ws.conditional_formatting.add(
    "B2:B21",
    CellIsRule(
        operator="lessThan",
        formula=["30"],
        fill=PatternFill(start_color="FFC7CE", end_color="FFC7CE", fill_type="solid"),
        font=Font(color="9C0006"),
    ),
)

# --- Formula Rule: highlight entire row if score > 90 ---
ws.conditional_formatting.add(
    "A2:D21",
    FormulaRule(
        formula=["$B2>90"],
        fill=PatternFill(start_color="DDEBF7", end_color="DDEBF7", fill_type="solid"),
    ),
)

wb.save("/tmp/conditional_formatting.xlsx")
```

### Data Validation

```python
from openpyxl import Workbook
from openpyxl.worksheet.datavalidation import DataValidation

wb = Workbook()
ws = wb.active
ws.title = "Data Entry"

# Headers
ws.append(["Department", "Amount", "Start Date", "Priority", "Email"])

# --- Dropdown list ---
dept_validation = DataValidation(
    type="list",
    formula1='"Sales,Marketing,Engineering,Finance,HR"',
    allow_blank=True,
)
dept_validation.error = "Please select a valid department."
dept_validation.errorTitle = "Invalid Department"
dept_validation.prompt = "Choose a department from the list."
dept_validation.promptTitle = "Department"
ws.add_data_validation(dept_validation)
dept_validation.add("A2:A100")

# --- Number validation (100 to 1000000) ---
amount_validation = DataValidation(
    type="whole",
    operator="between",
    formula1=100,
    formula2=1000000,
    allow_blank=False,
)
amount_validation.error = "Amount must be between 100 and 1,000,000."
amount_validation.errorTitle = "Invalid Amount"
amount_validation.prompt = "Enter a whole number between 100 and 1,000,000."
amount_validation.promptTitle = "Amount"
ws.add_data_validation(amount_validation)
amount_validation.add("B2:B100")

# --- Date validation (2024-01-01 to 2025-12-31) ---
date_validation = DataValidation(
    type="date",
    operator="between",
    formula1="2024-01-01",
    formula2="2025-12-31",
    allow_blank=False,
)
date_validation.error = "Date must be between 2024-01-01 and 2025-12-31."
date_validation.errorTitle = "Invalid Date"
date_validation.prompt = "Enter a date in YYYY-MM-DD format (2024-2025)."
date_validation.promptTitle = "Start Date"
ws.add_data_validation(date_validation)
date_validation.add("C2:C100")

# --- Dropdown for priority ---
priority_validation = DataValidation(
    type="list",
    formula1='"Low,Medium,High,Critical"',
    allow_blank=False,
)
priority_validation.error = "Select a priority level."
priority_validation.errorTitle = "Invalid Priority"
ws.add_data_validation(priority_validation)
priority_validation.add("D2:D100")

# --- Text length validation (email max 50 chars) ---
email_validation = DataValidation(
    type="textLength",
    operator="lessThanOrEqual",
    formula1=50,
    allow_blank=True,
)
email_validation.error = "Email must be 50 characters or fewer."
email_validation.errorTitle = "Too Long"
ws.add_data_validation(email_validation)
email_validation.add("E2:E100")

# Add sample data
ws.append(["Sales", 5000, "2024-06-15", "High", "user@example.com"])

wb.save("/tmp/data_validation.xlsx")
```

### Freeze Panes & Auto-Filter

```python
from openpyxl import Workbook

wb = Workbook()

headers = ["ID", "Region", "Product", "Units", "Revenue", "Date"]

base_date = datetime.date(2024, 1, 1)

    ws.append([
        i,
        random.choice(regions),
        random.choice(products),
        random.randint(10, 500),
```

### Read Existing File

```python
from openpyxl import load_workbook

# --- Read with formulas as strings ---
wb = load_workbook("/tmp/styled_report.xlsx")
print("Sheets:", wb.sheetnames)

ws = wb.active
print(f"Active sheet: {ws.title}")
print(f"Dimensions: {ws.dimensions}")

# Iterate all rows
for row in ws.iter_rows(min_row=1, max_row=ws.max_row, values_only=True):
    print(row)

# Access specific cell
print(f"Cell B2 = {ws['B2'].value}")

wb.close()

# --- Read with calculated values (data_only) ---
wb2 = load_workbook("/tmp/styled_report.xlsx", data_only=True)
ws2 = wb2.active

# data_only=True returns cached calculated values instead of formulas
# Note: values are from the last time the file was opened in Excel
for row in ws2.iter_rows(min_row=1, max_col=5, max_row=3, values_only=True):
    print(row)

wb2.close()

# --- Read specific range ---
wb3 = load_workbook("/tmp/styled_report.xlsx")
ws3 = wb3.active
for row in ws3["A1":"E3"]:
    for cell in row:
        print(f"{cell.coordinate}: {cell.value} (font={cell.font.name}, bold={cell.font.bold})")

wb3.close()
```

```python
from docx import Document
from docx.shared import Pt, RGBColor
from docx.oxml.ns import qn
from docx.oxml import OxmlElement
import docx

def add_hyperlink(paragraph, text, url):
    """Add a hyperlink to a paragraph (low-level XML approach)."""
    part = paragraph.part
    r_id = part.relate_to(
        url,
        docx.opc.constants.RELATIONSHIP_TYPE.HYPERLINK,
        is_external=True,
    )

    hyperlink = OxmlElement("w:hyperlink")
    hyperlink.set(qn("r:id"), r_id)

    new_run = OxmlElement("w:r")
    rPr = OxmlElement("w:rPr")

    # Blue underlined text (standard hyperlink style)
    color = OxmlElement("w:color")
    color.set(qn("w:val"), "0563C1")
    rPr.append(color)

    underline = OxmlElement("w:u")
    underline.set(qn("w:val"), "single")
    rPr.append(underline)

    new_run.append(rPr)
    new_run.text = text
    hyperlink.append(new_run)
    paragraph._p.append(hyperlink)

    return hyperlink

doc = Document()
doc.add_heading("Resources", level=1)

# Hyperlink in its own paragraph

add_hyperlink(para, "https://example.com", "https://example.com")

# Hyperlink with custom text
para2 = doc.add_paragraph("For documentation, see the ")
add_hyperlink(para2, "User Guide", "https://docs.example.com/guide")
para2.add_run(".")

# Multiple hyperlinks in one paragraph
para3 = doc.add_paragraph("Follow us on ")
add_hyperlink(para3, "Twitter", "https://twitter.com/example")
para3.add_run(" and ")
```
